ternurines = ['Freya Chocolate', 
                'Sharon Fennec', 
                'Kippie Waddle', 
                'Ginger Snowdrift', 
                'Kate Perlwinkle', 
                'Avril Husky', 
                'Nancy Brightfield', 
                'Bran Highbranch', 
                'Lona Dale',

function randomTernurin() {
    let random = Math.floor(Math.random() * ternurines.length);
    return ternurines[random];
}
